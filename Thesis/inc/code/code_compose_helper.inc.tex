services:
  db:
    image: mysql
    environment:
      - MYSQL_ROOT_PASSWORD=yes
    ports:
      - 3306:3306
    volumes:
      - sql_data:/var/lib/mysql
  mongo:
    image: mongo
    environment:
      - MONGO_INITDB_ROOT_USERNAME=root
      - MONGO_INITDB_ROOT_PASSWORD=none
    ports:
      - 27017:27017
    volumes:
      - mongo_data:/data/db
  mosquitto:
    image: eclipse-mosquitto
    ports:
      - 1883:1883
    volumes:
      - mosquitto_data:/mosquitto/data

  mongo_data:
